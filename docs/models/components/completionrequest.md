# CompletionRequest

## Example Usage

```typescript
import { CompletionRequest } from "atoma-sdk/models/components";

let value: CompletionRequest = {
  frequencyPenalty: 0,
  logitBias: {
    "1234567890": 0.5,
    "1234567891": -0.5,
  },
  logprobs: 1,
  model: "meta-llama/Llama-3.3-70B-Instruct",
  n: 1,
  presencePenalty: 0,
  prompt: [
    "Hello!",
  ],
  seed: 123,
  stop: [
    "json([\"stop\", \"halt\"])",
  ],
  stream: false,
  suffix: "json(\"\n\")",
  temperature: 0.7,
  topP: 1,
  user: "user-1234",
};
```
